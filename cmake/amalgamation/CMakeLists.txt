file(GLOB_RECURSE HEADER_FILES CONFIGURE_DEPENDS "${PROJECT_SOURCE_DIR}/include/*.hpp")
file(GLOB_RECURSE SOURCE_FILES CONFIGURE_DEPENDS "${PROJECT_SOURCE_DIR}/src/*.cpp")
set(AMALGAMATED_FILE "${SAFETYHOOK_AMALGAMATED_DIR}/safetyhook.cpp")
set(AMALGAMATED_HEADER "${SAFETYHOOK_AMALGAMATED_DIR}/safetyhook.hpp")
set(AMALGAMATE_SCRIPT "${PROJECT_SOURCE_DIR}/amalgamate.py")

add_custom_command(
    OUTPUT "${AMALGAMATED_FILE}" "${AMALGAMATED_HEADER}"
    DEPENDS ${HEADER_FILES} ${SOURCE_FILES} "${AMALGAMATE_SCRIPT}"
    COMMAND "${Python3_EXECUTABLE}" "${AMALGAMATE_SCRIPT}" --output-dir "${SAFETYHOOK_AMALGAMATED_DIR}"
    MAIN_DEPENDENCY "${AMALGAMATE_SCRIPT}"
    COMMENT "Amalgamating"
)

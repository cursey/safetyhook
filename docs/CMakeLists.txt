file(GLOB_RECURSE HEADER_FILES CONFIGURE_DEPENDS "${PROJECT_SOURCE_DIR}/include/*.hpp")
set(DOXYGEN_INDEX_FILE "${PROJECT_SOURCE_DIR}/docs/html/index.html")
set(DOXYFILE_IN "${CMAKE_CURRENT_SOURCE_DIR}/Doxyfile.in")
set(DOXYFILE_OUT "${CMAKE_CURRENT_SOURCE_DIR}/Doxyfile")

configure_file("${DOXYFILE_IN}" "${DOXYFILE_OUT}")
add_custom_command(
    OUTPUT "${DOXYGEN_INDEX_FILE}"
    DEPENDS ${HEADER_FILES} "${DOXYFILE_IN}"
    COMMAND "${DOXYGEN_EXECUTABLE}" "${DOXYFILE_OUT}"
    MAIN_DEPENDENCY "${DOXYFILE_OUT}"
    COMMENT "Generating docs"
)
add_custom_target(Doxygen ALL DEPENDS "${DOXYGEN_INDEX_FILE}")
